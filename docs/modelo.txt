usuario
    - id
    - nombre
    - apellido
    - email
    - createdAt

pregunta
    - id
    - dificultad (tinyInt: 1=facil, 2=medio, 3=dificil)
    - texto
    - createdAt

respuesta
    - id
    - preguntaId
    - texto
    - esCorrecta (boolean)
    - createdAt

trivia
    - id
    - nombre
    - descripcion
    - createdAt

preguntaTrivia
    - triviaId
    - preguntaId
    - orden (integer)
    - createdAt

usuarioTrivia
    - id
    - triviaId
    - usuarioId
    - puntajeTotal (integer, nullable)
    - estado (tinyInt: 0=no iniciado, 1=iniciado, 2=finalizado) - default 0
    - codigo (Código único para invitar al usuario a responder al trivia vía url) - length 6 characters - uppercase - alphanumeric
    - createdAt
    - startedAt
    - finishedAt

respuestaUsuario
    - id
    - usuarioTriviaId
    - preguntaId
    - respuestaId
    - esCorrecta (boolean)
    - puntajeObtenido (integer)
    - createdAt
